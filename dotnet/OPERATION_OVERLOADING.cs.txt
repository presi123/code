using System;
namespace OPERATION_OVERLOADING
{
    class complex
    {
        private int x;
        private int y;
        public complex()
        {

        }
        public complex(int i, int j)
        {
            x = i;
            y = j;
        }
        public void showxy()
        {
            Console.WriteLine("{0} {1}",x,y);
        }
        public static complex operator +(complex c1, complex c2) 
        {
            complex temp = new complex();
            temp.x=c1.x + c2.x;
            temp.y=c1.y + c2.y;
            return temp;

        }
    }
    internal class Program
    {
        static void Main(string[] args)
        {
            complex c1=new complex(12,13);
            complex c2=new complex(10,23);
            complex c3=new complex();
            c3 = c1 + c2;
            Console.WriteLine("X and Y values of c1 object are :");
            c1.showxy();
            Console.WriteLine("X and Y values of c2 object are :");
            c2.showxy();
            Console.WriteLine("X and Y values of c3 object are :");
            c3.showxy();
        }
    }
}
