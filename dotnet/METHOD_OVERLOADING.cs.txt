namespace Method_overloading
{
    class Program
    {
        public void display(int i)
        {
            Console.WriteLine("parameter-integer" + i + "is called");
        }
        public void display(string n)
        {
            Console.WriteLine("function with string" + n + "is called");

        }
        public void display(string n, int i)
        {
            Console.WriteLine("function with string" +  n  + "and" + i + "integer is called");

        }
    }
    class main
    {
        static void Main(string[] args)
        {
            Program p= new Program();
            p.display(10);
            p.display("waseem");
            p.display("waseem", 10);
        }

    }
       
    
}
